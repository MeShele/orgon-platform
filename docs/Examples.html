Пример GET запроса¶

// Импортируем необходимые модули
const Web3 = require('web3');
const axios = require('axios');

// Асинхронная функция для выполнения запроса к API
(async function testAPI() {
    // Подпись данных (в данном случае пустого объекта) с использованием приватного ключа
    const web3 = new Web3();
    const privateKey = '0xb7d9b51eda54d664366d74c1b1a4ed2ca9e1abed646732bfb83b141f22df3b39';
    const dataToSign = JSON.stringify({});
    const sign = await web3.eth.accounts.sign(dataToSign, privateKey);

    // Заголовки для запроса, включающие адрес и подпись
    const headers = {
        'x-app-ec-from': '0xA285990a1Ce696d770d578Cf4473d80e0228DF95',
        'x-app-ec-sign-r': sign.r,
        'x-app-ec-sign-s': sign.s,
        'x-app-ec-sign-v': sign.v
    };

    // Выполнение GET-запроса к API с указанными заголовками
    const response = await axios.get('https://my.h2k.me/ece/netlist/0', { headers });
})();

Пример POST запроса по созданию кошелька¶

// Импортируем необходимые модули
const Web3 = require('web3');
const axios = require('axios');

// Асинхронная функция для выполнения запроса к API
(async function createNewWallet() {
    // Подпись данных (в данном случае объекта с информацией о кошельке) с использованием приватного ключа
    const web3 = new Web3();
    const privateKey = '0xb7d9b51eda54d664366d74c1b1a4ed2ca9e1abed646732bfb83b141f22df3b39';
    const requestBody = {
        "slist": {
            "min_signs": "2",
            "0": { "type": "all", "email": "e@mail.com" },
            "1": { "type": "any", "email": "e@mail.ru", "sms": "+77777777777" },
            "2": { "type": "all", "ecaddress": "0xa75db3c448bb62e208e4babf500925427837e464" }
        },
        "network": "1000",
        "info": "Мой первый кошелёк" 
    };
    const dataToSign = JSON.stringify(requestBody);
    const sign = await web3.eth.accounts.sign(dataToSign, privateKey);

    // Заголовки для запроса, включающие адрес и подпись
    const headers = {
        'x-app-ec-from': '0xA285990a1Ce696d770d578Cf4473d80e0228DF95',
        'x-app-ec-sign-r': sign.r,
        'x-app-ec-sign-s': sign.s,
        'x-app-ec-sign-v': sign.v,
        'Content-Type': 'application/json'
    };

    // Выполнение POST-запроса к API с указанными заголовками и телом запроса
    const response = await axios.post('https://my.h2k.me/ece/newWallet', requestBody, { headers });
})();
